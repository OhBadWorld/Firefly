# 使用 Langchain.js 实现一个多工具 Agent，能够根据用户査询调用不同的工具(如天气査询、计算器、搜索等)，并返回整合的结果。要求设计清晰的 Agent 架构和工具调用流程。

考察点

- Agent 架构设计(LLM+Planning+Memory+Tools Calling)
- ReAct(思考-行动-观察循环)
- 自定义工具-
- 多工具协调并且结果整合

## 解题思路

Agent 的核心架构可以总结为"1 个核心 + 3个支撑"，即 Agent = LLM(核心驱动) + 规划(Planning) + 记忆(Hemory) + 工具使用(Tool Use) 在LangChain.js 中，Agent 的执行遵循 ReAct 框架的"思考 -行动 - 观察”循环机制。

用户输入问题后，Agent 会将问题、可用工具列表和 ReAct指令模板组合成初始提示，发送给LLM。LLM 根据当前信息进行思考(Think)，输出下一步的行动计划。然后执行行动(Act)，调用相应的工具并获取观察结果(Observation)。这个过程会循环进行，直到 LLM 判断已经得到最终答案。

在工具设计方面，需要创建自定义工具类。每个工具都应该继承自 BaseTool类，实现\_run 同步方法和\_arun 异步方法。工具需要定义 name(工具名称)和description(工具描述)，这些信息会用于LLM 的决策过程。例如，天气查询工具需要接收地理位置和时间参数，调用天气API 获取结果;计算器工具则执行数学运算。

langchain 其实屏蔽了这个复杂度，需要设计合理的工具选择策略，让 LLM 能够根据问题类型选择合适的工具。可以建立工具的分类体系，如信息查询类、计算类、操作类等，并为每个工具设置适用的问题模式。同时要处理工具调用的依赖关系，有些复杂问题可能需要多个工具的协作才能解决。

设计不同的整合策略，如简单拼接、优先级排序、内容摘要等。对于需要进一步处理的结果，如计算结果的格式化、查询结果的筛选等，需要在 Agent 中实现相应的处理逻辑。

> tips 考察大家在agent核心架构层 和 reAct循环机制层的积累，自定义工具会不会定义，多工具的整合方法

## 知识点

- Agent 核心架构和 ReAct 循环机制
- 自定义工具的开发规范
- 工具选择和协调策略
- 多工具结果整合方法
- LangChain.js的 Agent 实现原理

#### 1)·自定义工具实现

```typescript
// lib/langchain/tools.ts
import { tool } from '@langchain/core/tools';
import { z } from 'zod';

// 1.计算器工具
export const calculatorTool =tool(
    async({expression}){
        try {
            // 安全的表达式计算
            const result = Function(`"use strict"; return(${expression})`)();
            return `计算结果:${result}`;
        } catch (error){
            return  `计算错误:${(error as Error).message}`;
        },
    },
    {
        name:'calculator',
        description:'执行数学表达式计算。支持加减乘除、括号、基本数学函数。',
        schema:z.object({
            expression:z.string().describe('要计算的数学表达式，例如:"25 *4 + 100"'),
        }),

    }
);

// 2.天气查询工具
export const weatherTool = tool(
  async({ city,date }) => {
    // 这里调用真实的天气API
    const mockWeather:Record<string,any> = {
        '北京':{temp:25，condition:'晴'，humidity:45 },
        '上海':{temp:28，condition:'多云'，humidity:60 },
        '深圳':{temp:30，condition:'雷阵雨'，humidity:75 },
    };

    const weather = mockWeather[city] || { temp: 22, condition:'未知', humidity: 50 };
    return `${date || '今天'} ${city}的天气: ${weather.condition}, 温度 ${weather.temp}°c, 湿度${weather.humidity%`;
  },
  {
    name: 'get_weather',
    description:'查询指定城市的天气信息。',
    schema: z.object({
        city:z.string().describe('城市名称，例如:"北京"),
        date:z.string().optional().describe('查询日期，格式:YYYY-MM-DD，默认为今天'),
    }),
  }
);

// 3.知识库检索工具(RAG 集成)
export const knowledgeBaseTool =tool(
    async({ query, topK=3 }) => {
      // 这里集成真实的向量检索
      const mockDocs =[
        { content: '公司成立于 2010 年，总部位于北京'，source:'company-profile.pdf'},
        { content: '产品支持 7 天无理由退换货，质保期1年'，source:'after-sales.md'},
        { content:"工作时间为周一至周五 9:00-18:00，法定节假日休息'，source:'hr-policy.md'},
      ];

      return mockDocs.slice(0，topK).map((doc，i) => `[${i +1}]${doc.content}(来源:${doc.source})`)join('\n');
    },
    {
        name:'search_knowledge_base',
        description:'搜索公司内部知识库，查询产品信息、售后政策、公司制度等相关内容。',
        schema: z.object({
            query:z.string().describe('搜索查询词')
            topK:z.number().optional().describe('返回结果数量，默认 3条'),
        }),
    }
);
// 导出工具集
export const allTools =[calculatorTool, weatherTool, knowledgeBaseTool]

```

#### 2) 使用 LangGraph 构建 ReActAgent

```typescript
// lib/langchain/agent.ts
import { StateGraph, END, START, MemorySaver } from '@langchain/langgraph';
import { ToolNode } from '@langchain/langgraph/prebuilt';
import { chatPromptTemplate, MessagesPlaceholder } from '@langchain/core/prompts';
import { HumanMessage, AIMessage, BaseMessage } from '@langchain/core/messages';
import { allTools }from './tools';
import { llm } from './setup';

// 1、绑定工具到 LLM
const llmWithTools =llm.bindTools(allTools);
const toolNode =new ToolNode(allTools);
// 2.定义 Agent 状态
interface AgentState {
    messages: BaseMessage[];
}

// 3.定义 Agent 节点
async function agentNode(state: AgentState){
    const{ messages } = state;
    const prompt =ChatPromptTemplate.fromMessages([
        [
            'system',
            '你是一个有用的 AI 助手。你可以使用以下工具来帮助用户:\n' + ,
            '- calculator:数学计算\n' +
            '- get_weather:天气查询\n'+
            '- search_knowledge_base:知识库搜索\n\n' +
            '请根据用户的问题选择合适的工具。如果不需要工具，直接回答。'
        ],
        new MessagesPlaceholder('messages')
    ]);

    const chain =prompt.pipe(llmWithTools);
    const response =await chain.invoke({messages }):

    return { messages:[response]};
}

// 4、路由函数:判断是否需要调用工具
function shouldContinue(state: AgentState): 'tools'| typeof END {
    const lastMessage =state.messages[state.messages.length - 1] as AIMessage;
    if(lastMessage.tool calls?.length){
        return 'tools';
    }
    return END;
}

// 5.构建状态图
const workflow = new StateGraph<AgentState>({
    channels:{
        messages:{
            value: (x:BaseMessage[]，y:BaseMessage[]) => x.concat(y),
            default:() => [],
        },
    },
})
.addNode('agent', agentNode)
.addNode('tools', toolNode)
.addEdge(START, 'agent')
.addConditionalEdges('agent', shouldContinue)
.addEdge('tools','agent');

// 从start -> agent -> tool -> (tool没调完，继续调tool shouldContinue 循环) -> 调完后回到 agent

// 6. 编译并添加记忆
const checkpointer =new MemprySaver();
export const agent =workflow.compile({ checkpointer });


// 7、使用示例
export async function runAgent(userInput: string, threadId: string = 'default'){
    const config={configurable:{ thread_id: threadId } };
    const result =await agent.invoke(
        { messages: [new HumanMessage(userInput)]}
        config,
    );
    const finalMessage =result.messages[result.messages.length - 1];
    return finalMessage.content as string;
}
```

#### 3)Agent 使用示例

```typescript
// 使用 Agent
async function main() {
  // 计算问题
  console.log(await runAgent("请计算 156 *23 + 789 等于多少"));
  // 天气问题
  console.log(await runAgent("北京今天天气怎么样?"));
  // 知识库问题
  console.log(await runAgent("公司的退换货政策是什么?"));
  // 多轮对话(自动记住上下文)
  console.log(await runAgent("那上海呢?")); // 自动知道问的是天气
}
```
